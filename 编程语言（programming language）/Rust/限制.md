### 契约限制

```rust
fn main() {
    let string = String::from("example");
    let borrowed = "hello";
    let cow = std::borrow::Cow::Borrowed("world");

    let re1 = frobnicates3::<&str>(string.as_ref());
    let re2 = frobnicates3::<&str>(borrowed);
    let re3 = frobnicates3(cow);

    println!("re1: {:?}", re1);
    println!("re2: {:?}", re2);
    println!("re3: {:?}", re3);
}

fn frobnicates3<T: AsRef<str>>(s: T) -> T {
    s
}

```

### 泛型参数

两份函数拷贝

```rust
fn main() {
    let s = String::from("hello");
    let r = "word";
    print_as_str(s);
    print_as_str(r)
}

fn print_as_str<T: AsRef<str>>(s: T) {
    println!("{}", s.as_ref());
}

```

一份函数拷贝

```rust
fn main() {
    let s = String::from("hello");
    let r = "word";
    print_as_str(&s);
    print_as_str(&r);
}

fn print_as_str(s: &dyn AsRef<str>) {
    println!("{}", s.as_ref());
}

```
